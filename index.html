function formatDate(val) { if (!val) return ''; var d = new Date(val); if (isNaN(d)) return String(val); var y = d.getFullYear(); var m = String(d.getMonth()+1).padStart(2,'0'); var day = String(d.getDate()).padStart(2,'0'); return y + '-' + m + '-' + day; } function formatTime(val) { if (!val) return ''; if (typeof val === 'string' && val.includes(':')) return val.substring(0,5); var d = new Date(val); if (isNaN(d)) return String(val); var h = String(d.getHours()).padStart(2,'0'); var min = String(d.getMinutes()).padStart(2,'0'); return h + ':' + min; } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); sheet.appendRow([data.date, data.time, data.invoice, data.amount, data.salesman, data.type, data.paid]); return ContentService .createTextOutput(JSON.stringify({result: 'success'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rows = sheet.getDataRange().getValues(); var data = []; for (var i = 1; i < rows.length; i++) { var row = rows[i]; if (!row[0]) continue; data.push({ date: formatDate(row[0]), time: formatTime(row[1]), invoice: String(row[2]), amount: row[3], salesman: String(row[4]), type: String(row[5]), paid: String(row[6]) }); } var json = JSON.stringify(data); var callback = e.parameter.callback; if (callback) { return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); }
